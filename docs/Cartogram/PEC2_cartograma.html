Cargar datos

data("World")

Continuous Area Cartogram

# project the map
world_map <- st_transform(World, 3395)

# construct cartogram
world_continuous_area <- cartogram_cont(world_map, "life_exp", itermax = 5)
## Warning in cartogram_cont.sf(world_map, "life_exp", itermax = 5): NA not
## allowed in weight vector. Features will be removed from Shape.

Visualización

ggplot(world_continuous_area) +
  geom_sf(aes(fill = life_exp ), linewidth = 0, alpha = 0.9) +
  theme_void() +
  scale_fill_viridis_c(
    name = "Esperanza (Años)",
    breaks = c(50, 60, 70, 80),
    direction = -1,
    guide = guide_legend(
      keyheight = unit(3, units = "mm"),
      keywidth = unit(12, units = "mm"),
      label.position = "bottom",
      title.position = "top", nrow = 1
    )
  ) +
  labs(title = "Esperanza de vida") +
  theme(
    text = element_text(color = "#22211d"),
    plot.background = element_rect(fill = "#f5f5f4", color = NA),
    panel.background = element_rect(fill = "#f5f5f4", color = NA),
    legend.background = element_rect(fill = "#f5f5f4", color = NA),
    plot.title = element_text(
      size = 22, hjust = 0.5,
      color = "#4e4d47",
      margin = margin(
        b = -0.1, t = 0.4, l = 2,
        unit = "cm"
      )
    ),
    legend.position = "bottom"
  )
